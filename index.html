fig1.1 <- xlsx_wide |> 
   left_join(select(trials, drugs, id), by = "id") |> 
  mutate(greater = ifelse(Experimental > Control, "experi", "control"), 
         event = paste0(drugs, ", ", adverse, ", ", starified)) |> 
 
  ggplot(aes(y = Experimental, x = Control, color = greater, key = event)) +
    geom_abline(intercept = 0, color = "grey", linewidth = 1) +
  geom_point(alpha = 0.55) +
  scale_color_manual(values  = c( "#373F51", "#E49273"), 
                     aesthetics = "color") +
  scale_y_percent(limits = c(0,1))+
  scale_x_percent(limits = c(0,1))+
  labs(title = "Adverse Events, all reported") +
  theme_minimal(base_family = "Roboto Condensed") +
  theme(legend.position = "none", 
        # axis.line.x = element_line(color = "red"), 
        axis.ticks.x = element_blank())
